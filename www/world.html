require("mapmisc", quietly = TRUE)
## map images will be cached in  /tmp/mapmiscCache_patrick
worldLL = openmap(extent(-180, 180, -80, 80), zoom = 1, crs = mapmisc::crsLL)


Scentre = c(Grenwich = 0, Toronto = -79, `Hong kong` = 114)

mollList = mapply(mapmisc::moll, x = Scentre)


worldMoll = mapply(mapmisc::openmap, x = mollList, MoreArgs = list(zoom = 1, 
    fact = 2))
map.new(worldLL)
plot(worldLL, add=TRUE)

for(D in names(worldMoll)) {
  map.new(mollList[[D]])
  plot(worldMoll[[D]], add=TRUE)
}
